apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: gke-ingress-service
  annotations:
    # Associate the Ingress with your static IP
    kubernetes.io/ingress.global-static-ip-name: "gke-static-ip"
    # Associate the Ingress with your managed certificate
    networking.gke.io/managed-certificates: "gke-managed-cert"
spec:
  rules:
  - host: your-app.com # <-- Add your domain name here
    http:
      paths:
        - path: /api
          pathType: Prefix
          backend:
            service:
              name: server-cluster-ip-service
              port:
                number: 5000
        - path: /
          pathType: Prefix
          backend:
            service:
              name: client-cluster-ip-service
              port:
                number: 3000
